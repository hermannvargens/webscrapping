document.getElementById('finalize-purchase-button').addEventListener('click', function() { // Capturar o conteúdo da seção "Resumo das Compras" const summarySection = document.getElementById('summary-content').innerHTML; // Criar o conteúdo do arquivo HTML const htmlContent = ` Resumo das Compras
${summarySection}
`; // Criar um blob com o conteúdo HTML const blob = new Blob([htmlContent], { type: 'text/html' }); // Criar um link temporário para download const link = document.createElement('a'); link.href = URL.createObjectURL(blob); link.download = 'resumo_compras.html'; // Simular um clique no link para iniciar o download link.click(); // Liberar o objeto URL URL.revokeObjectURL(link.href); });
